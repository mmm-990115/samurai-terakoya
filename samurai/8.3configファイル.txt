8.3のconfigファイル


R1(config)#int g0/0											g0/0を指定し特定インタフェースコンフィグレーションモードに移行
R1(config-if)#ip address 192.168.1.1 255.255.255.0			IPアドレスを設定
R1(config-if)#no shutdown									g0/0を有効化
R1(config-if)#exit											グローバルコンフィグレーションモードに移行


R1(config)#int s0/0/0										s0/0/0を指定し特定インタフェースコンフィグレーションモードに移行
R1(config-if)#ip address 10.0.10.5 255.255.255.252			IPアドレスを設定
R1(config-if)#no shutdown									s0/0/0を有効化
R1(config-if)#clock rate 64000								クロック信号の設定
R1(config-if)#exit											グローバルコンフィグレーションモードに移行


R1(config)#router ospf 1									OSPF1を指定し特定インタフェースコンフィグレーションモードに移行
R1(config-router)#network 192.168.1.0 0.0.0.255 area 0		OSPFの設定(g0/0の所属するネットワークに対して)
R1(config-router)#network 10.0.10.4 0.0.0.3 area 0			OSPFの設定(s0/0/0の所属するネットワークに対して)
R1(config-router)#exit										グローバルコンフィグレーションモードに移行


R1(config)#ip access-list extended FTP_block				拡張アクセスコントロールリスト「FTP_block」を作成
R1(config-ext-nacl)#deny tcp 192.168.1.2 0.0.0.0 192.168.200.2 0.0.0.0 eq 21
															TCPを用いた192.168.1.2から192.168.200.2に対してFTPに関する通信を拒否する
※deny tcp host 192.168.1.2 host 192.168.200.2 eq 21 でも可
R1(config-ext-nacl)#permit ip any any						全ての通信で全てのIPアドレスから全てのIPアドレスに関する通信を許可する
R1(config-ext-nacl)#exit											グローバルコンフィグレーションモードに移行


R1(config)#int g0/0											g0/0を指定し特定インタフェースコンフィグレーションモードに移行
R1(config-if)#ip access-group FTP_block in					ACL「FTP_block」をinboundに適用


【R2】
Router>enable												特権モードに移行
Router#configure terminal									グローバルコンフィグレーションモードに移行 (conf t でも可)
Router(config)#hostname R2									名前をR2に変更
R2(config)#enable password cisco							enable パスワードの設定
R2(config)#enable secret kenit								enable secretパスワードの設定
R2(config)#line console 0									コンソールを指定し特定インタフェースコンフィグレーションモードに移行
R2(config-line)#password conKEN								コンソールパスワードの設定
R2(config-line)#login										パスワードの有効化
R2(config-line)#line vty 0 4								vty回線を指定し特定インタフェースコンフィグレーションモードに移行
R2(config-line)#password vtyKEN								vtyパスワードの設定
R2(config-line)#transport input telnet						telnetの使用を有効化
R2(config-line)#exit										グローバルコンフィグレーションに移行


R2(config)#int g0/0											g0/0を指定し特定インタフェースコンフィグレーションモードに移行
R2(config-if)#ip address 192.168.200.1 255.255.255.0		IPアドレスを設定
R2(config-if)#no shutdown									g0/0を有効化
R2(config-if)#exit											グローバルコンフィグレーションに移行


R2(config)#int s0/0/0										s0/0/0を指定し特定インタフェースコンフィグレーションモードに移行
R2(config-if)#ip address 10.0.0.6 255.255.255.252			IPアドレスを設定
R2(config-if)#no shutdown									g0/1を有効化
R2(config-if)#exit											グローバルコンフィグレーションに移行


R2(config-if)#router ospf 1									OSPF1を指定し特定インタフェースコンフィグレーションに移行
R2(config-router)#network 192.168.200.0 0.0.0.255 area 0	OSPFの設定(g0/0の所属するネットワークに対して)
R2(config-router)#network 10.0.0.4 0.0.0.3 area 0			OSPFの設定(s0/0/0の所属するネットワークに対して)
R2(config-router)#exit										グローバルコンフィグレーションモードに移行


R2(config)#access-list 1 permit host 192.168.200.2			ACL「1」に192.168.200.2からの通信を許可する
※access-list 1 permit 192.168.200.2 0.0.0.0 でも可
R2(config)#line vty 0 4										vty回線を指定し特定インタフェースコンフィグレーションモードに移行
R2(config-line)#access-class 1 in							ACL「１」をinboundに適用
